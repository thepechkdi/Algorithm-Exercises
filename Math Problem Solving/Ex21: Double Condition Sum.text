BEGIN
    WRITE "Enter n:"
    READ n

    S = 0

    FOR i = 1 TO n DO

        IF (i MOD 2 = 0) AND (i MOD 3 = 0) THEN
            S = S + i * i
        ELSE IF (i MOD 2 = 0) OR (i MOD 3 = 0) THEN
            S = S + i
        ELSE
            // add 0, do nothing
        END IF

    END FOR

    WRITE "Sum S =", S
END
