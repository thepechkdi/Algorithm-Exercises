Algorithm Average_N_Numbers
Var N, i : Integer
Var number, sum, average : flaot

Begin
   Write("Enter how many numbers you want: ")
   Read(N)
   sum = 0

   For i = 1 to N do
       Write("Enter number ", i, ": ")
       Read(number)
       sum = sum + number
   EndFor

   average = sum / N
   Write("The average is: ", average)
End
